Attribute für Dokumentklassen (Vererbung gibt's ja quasi schon):

\newcommand \newattribute[1] {
    \newcommand {\expandafter \csname akk@attribute@#1 \endcsname} {}
    \newcommand {\expandafter \csname set#1 \endcsname} {\renewcommand \expandafter \csname akk@attribute@#1 \endcsname}
    \newcommand {\expandafter \csname get#1 \endcsname} {\expandafter \csname akk@attribute@#1 \endcsname}
}

\newattribute{Width}
\setWidth{30}
\getWidth
